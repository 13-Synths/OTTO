Struct otto::engines::EngineDispatcher

template <otto::engines::EngineType ET>
struct EngineDispatcher
{
    enum class ErrorCode;

    using exception = util::as_exception<ErrorCode>;

    static constexpr otto::engines::EngineType const type = ET;

    EngineDispatcher();

    void init();

    Engine<ET>& current() noexcept;

    Engine<ET> const& current() const noexcept;

    Engine<ET>& operator*() noexcept;

    Engine<ET> const& operator*() const noexcept;

    Engine<ET> const* operator->() const noexcept;

    Engine<ET>* operator->() noexcept;

    Engine<ET>& select(Engine<ET>* ptr);

    Engine<ET>& select(Engine<ET>& ref);

    Engine<ET>& select(std::size_t idx);

    Engine<ET>& select(std::string const& name);

    std::vector<EnginePatch> make_patches() const;

    Engine<ET>& apply_patch(otto::engines::EnginePatch const& seq);

private:
    std::vector<std::unique_ptr<Engine<ET> >> _engines;

    Engine<ET>* _current;
};

Function apply_patch

Engine<ET>& apply_patch(otto::engines::EnginePatch const& seq);

Apply an EnginePatch to a matching engine.

Effects:
If p.type == type, find an engine ´e´ for which e.name() == p.name, and apply the patch (e.from_json(p.data)).
Throws:
If no matching engine was found, exception is thrown, with the appropriate error code.
Returns:
A reference to the engine that matched the patch
