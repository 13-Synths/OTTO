class ByteFile
{
public:
    struct Error;

    using Position = int;

    using Path = filesystem::path;

    struct Chunk;

    Path path;

    ByteFile();

    ByteFile(otto::util::ByteFile::Path const&);

    ByteFile(otto::util::ByteFile&) = delete;

    ByteFile(otto::util::ByteFile&&);

    virtual ~ByteFile();

    otto::util::ByteFile& operator=(otto::util::ByteFile);

    friend void swap(otto::util::ByteFile&, otto::util::ByteFile&);

    void open(otto::util::ByteFile::Path const&);

    void close();

    void flush();

    bool is_open() const;

    virtual void create_file();

    virtual void write_file();

    Position position();

    result<void, std::streamsize> read_bytes(OutIter, int);

    template <std::size_t N>
    result<void, std::streamsize> read_bytes(bytes<N>&);

    template <typename InIter, typename = std::enable_if<is_iterator_v<InIter,std::byte,std::input_iterator_tag>>
    void write_bytes(InIter, InIter);

    template <typename InIter, typename = std::enable_if<is_iterator_v<InIter,std::byte,std::input_iterator_tag>>
    void write_bytes(InIter, int);

    template <std::size_t N>
    void write_bytes(bytes<N> const&);

    template <typename F>
    std::enable_if_t<util::is_invocable_v<F, Chunk &>, void> for_chunks_in_range(otto::util::ByteFile::Position, otto::util::ByteFile::Position, F&& f);

protected:
    std::fstream fstream;
};
